   ```typescript
   import Anthropic from '@anthropic-ai/sdk';

   const VISION_MODEL = 'claude-haiku-4-5-20251001';
   const VISION_FALLBACK_MODEL = 'claude-sonnet-4-6';
   const VISION_PROMPT =
     "Describe this image factually in 2-4 sentences. Note key visual elements, " +
     "text content if any, and overall composition. Do not interpret or speculate — " +
     "just describe what's visible.";
   const MAX_DESCRIPTION_TOKENS = 300;

   let client: Anthropic | null = null;
   let warnedMissingKey = false;
   function getClient(): Anthropic | null {
     if (!process.env.ANTHROPIC_API_KEY) {
       if (!warnedMissingKey) {
         console.warn('ANTHROPIC_API_KEY missing — image descriptions disabled');
         warnedMissingKey = true;
       }
       return null;
     }
     if (!client) client = new Anthropic({ apiKey: process.env.ANTHROPIC_API_KEY });
     return client;
   }

   export async function extractImageDescription(
     buffer: Buffer,
     mimeType: string,
   ): Promise<string | null> {
     if (!IMAGE_MIMES.has(mimeType)) return null;
     const c = getClient();
     if (!c) return null;
     const base64 = buffer.toString('base64');
     for (const model of [VISION_MODEL, VISION_FALLBACK_MODEL]) {
       try {
         const res = await c.messages.create({
           model,
           max_tokens: MAX_DESCRIPTION_TOKENS,
           messages: [{
             role: 'user',
             content: [
               { type: 'image', source: { type: 'base64', media_type: mimeType, data: base64 } },
               { type: 'text', text: VISION_PROMPT },
             ],
           }],
         });
         const block = res.content.find((b: { type: string }) => b.type === 'text');
         if (block && block.type === 'text') {
           const text = (block as { text: string }).text.trim();
           if (text) return text;
         }
         return null;
       } catch (err) {
         console.warn(`extractImageDescription failed on ${model}:`, err);
         // try next model
       }
     }
     return null;
   }
   ```

|-----------|----------|-----------|-------------|-----------------|
| T-17-1-01-01 | DoS | POST /attachments image flood | mitigate | Existing Phase 17 mitigations carry: 20 MB cap per file + 10 req/min/user limit + 4-image-per-message cap. Vision call adds ≤2s/image latency; worst case 4×2s = 8s synchronous, acceptable. No new mitigation needed. |
| T-17-1-01-02 | Tampering / Prompt-Injection | description column → prepended to agent prompt (Plan 17.1-02) | mitigate | The description IS user-influenced (user uploads an image whose pixels can contain adversarial text the vision API will faithfully describe). The `--- USER ATTACHMENTS ---` block in Plan 17.1-02 makes the boundary explicit to agents. Same trust level as the user's prompt itself — accepted as part of the user input. The prompt-template instruction "Do not interpret or speculate — just describe what's visible" reduces but does not eliminate the risk; agents are independently instructed (groupContext in routes.ts:179) to treat user-provided content as untrusted. |
| T-17-1-01-03 | Information Disclosure | ANTHROPIC_API_KEY leaked to logs | mitigate | The SDK never logs the key. Our `console.warn('ANTHROPIC_API_KEY missing — image descriptions disabled')` log message names the env var but does NOT log its value. The catch-block `console.warn(\`extractImageDescription failed on ${model}:\`, err)` logs the SDK error which may include status code / response body — verify locally that error objects do not echo the key. The Anthropic SDK does not include the API key in error objects (verified in SDK source convention). |
| T-17-1-01-04 | Repudiation | No audit trail for vision spend | accept | V1 — cost telemetry deferred to 17.2 per CONTEXT specific-ideas section. The artifacts row's `description` field IS the receipt: present = call succeeded, NULL = call skipped or failed. |
| T-17-1-01-05 | EoP | Anthropic SDK shells out / executes user-controlled bytes | mitigate | Anthropic SDK is pure JS HTTP client (no shell-out, no eval, no native modules). Image bytes go over HTTPS only. try/catch + null-on-failure prevents API exceptions from bubbling into the route handler and 5xx-ing the upload. |
| T-17-1-01-06 | DoS | Vision API outage blocks uploads | mitigate | extractImageDescription returns null on timeout / API error / network failure. The route's Promise.all never rejects (both helpers swallow errors). Uploads succeed with description=NULL; agents see the textual fallback in the next plan. |
</threat_model>
