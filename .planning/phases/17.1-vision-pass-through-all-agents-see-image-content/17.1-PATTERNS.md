### `apps/web/lib/extract-attachment.ts` (modified — add `extractImageDescription`)

**Analog 1 (existing file, lines 31-61):** the existing dispatch + try/catch + EXTRACT_CAP + warn-on-failure shape is already correct. Add a sibling exported function `extractImageDescription(buffer, mimeType)` that the route calls separately for image mimes — do NOT mutate `extractAttachment` to also do vision. **Analog 2 (`apps/web/lib/litellm-client.ts:55-108`):** the AbortController-timeout + try/catch + custom `*Error` class shape that improve-prompt uses for LLM calls. Anthropic SDK has its own timeout but the LiteLLM client's "wrap throw, log, return-fallback-or-rethrow" shape is the project's house style.

**Code excerpt to model on (litellm-client.ts:55-83):**
```ts
const controller = new AbortController();
const timer = setTimeout(() => controller.abort(), timeoutMs);
try {
  const res = await fetch(...);
  if (!res.ok) {
    const errBody = await res.text().catch(() => '');
    throw new LiteLLMError(`...`);
  }
  // ...
} catch (err) {
  if (err instanceof LiteLLMError) throw err;
  if ((err as { name?: string }).name === 'AbortError') {
    throw new LiteLLMError(`timeout`, err);
  }
  throw new LiteLLMError(`request failed`, err);
} finally {
  clearTimeout(timer);
}
```

**For 17.1 — proposed shape:**
```ts
import Anthropic from '@anthropic-ai/sdk';

const VISION_MODEL = 'claude-haiku-4-5-20251001';
const VISION_FALLBACK_MODEL = 'claude-sonnet-4-6';
const VISION_PROMPT =
  "Describe this image factually in 2-4 sentences. Note key visual elements, " +
  "text content if any, and overall composition. Do not interpret or speculate — " +
  "just describe what's visible.";
const MAX_DESCRIPTION_TOKENS = 300;

let client: Anthropic | null = null;
let warnedMissingKey = false;
function getClient(): Anthropic | null {
  if (!process.env.ANTHROPIC_API_KEY) {
    if (!warnedMissingKey) {
      console.warn('ANTHROPIC_API_KEY missing — image descriptions disabled');
      warnedMissingKey = true;
    }
    return null;
  }
  if (!client) client = new Anthropic({ apiKey: process.env.ANTHROPIC_API_KEY });
  return client;
}

export async function extractImageDescription(
  buffer: Buffer,
  mimeType: string,
): Promise<string | null> {
  if (!IMAGE_MIMES.has(mimeType)) return null;
  const c = getClient();
  if (!c) return null;
  const base64 = buffer.toString('base64');
  for (const model of [VISION_MODEL, VISION_FALLBACK_MODEL]) {
    try {
      const res = await c.messages.create({
        model,
        max_tokens: MAX_DESCRIPTION_TOKENS,
        messages: [{
          role: 'user',
          content: [
            { type: 'image', source: { type: 'base64', media_type: mimeType, data: base64 } },
            { type: 'text', text: VISION_PROMPT },
          ],
        }],
      });
      const block = res.content.find(b => b.type === 'text');
      return block && block.type === 'text' ? block.text.trim() : null;
    } catch (err) {
      console.warn(`extractImageDescription failed on ${model}:`, err);
      // try fallback
    }
  }
  return null;
}
```

The Anthropic SDK content-block discrimination + the two-model fallback loop are net-new patterns; the surrounding "module-load env check + warn-once + null-on-failure" mirrors the litellm-client style. CONTEXT specifies "fallback to Sonnet if Haiku errors twice in a row" — interpret literally as "try Haiku, on error try Sonnet"; the planner can iterate to "twice on Haiku before falling to Sonnet" if UAT shows Haiku is flaky.

```ts
import { randomUUID } from 'node:crypto';

export async function sendToAgent(
  cfg: OpenClawBridgeConfig,
  message: string,
  imageAttachments?: Array<{ filename: string; mimeType: string; base64: string }>,
): Promise<{ text; runId; durationMs; costUsd; model } | null> {
  const escapedMessage = message.replace(/'/g, "'\\''");
  const sessionId = `console:${cfg.agentId}:${cfg.runId}`;

  // 17.1: write each image to /tmp on the agent host (NOT locally — the SSH
  // boundary means /tmp/console-vision-... paths must exist where openclaw runs).
  // Implementation detail: scp + ssh, or write via the same ssh session, or
  // base64-data-URI inline. Researcher must verify the OpenClaw CLI shape.
  const tempPaths: string[] = [];
  let imageFlags = '';
  if (imageAttachments && imageAttachments.length > 0) {
    for (const att of imageAttachments) {
      const ext = att.mimeType.split('/')[1] ?? 'bin';
      const tempPath = `/tmp/console-vision-${randomUUID()}.${ext}`;
      tempPaths.push(tempPath);
      // TODO(executor): pick one of:
      //   (a) Write via SSH heredoc: `ssh ${cfg.sshHost} "base64 -d > ${tempPath}" <<< ${att.base64}`
      //   (b) scp from local /tmp to remote /tmp (writes locally first)
      //   (c) inline base64 as data URI in --message (no temp file)
      // Then append the appropriate flag — likely:
      imageFlags += ` --image '${tempPath}'`;
    }
  }

  const openclawCmd = cfg.sudoUser
    ? `sudo -u ${cfg.sudoUser} openclaw agent --message '${escapedMessage}'${imageFlags} --session-id '${sessionId}' --agent main --json 2>/dev/null`
    : `openclaw agent --message '${escapedMessage}'${imageFlags} --session-id '${sessionId}' --agent main --json 2>/dev/null`;
  const cmd = `ssh ... "timeout 120 ${openclawCmd}"`;

  try {
    const { stdout } = await execAsync(cmd, ...);
    // ... existing parse logic unchanged ...
    return { text, runId, durationMs, costUsd, model };
  } finally {
    // Best-effort cleanup of temp files on the remote host.
    if (tempPaths.length > 0) {
      const cleanupCmd = `ssh ${cfg.sshHost} "rm -f ${tempPaths.join(' ')}"`;
      execAsync(cleanupCmd).catch((err) => log.warn({ err }, 'temp cleanup failed'));
    }
  }
}
```
