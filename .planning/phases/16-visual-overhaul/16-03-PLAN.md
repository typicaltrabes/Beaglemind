    A) Create `apps/web/app/api/runs/history/summary/route.ts`:
    ```typescript
    import { NextResponse } from 'next/server';
    import { requireTenantContext, getTenantDb } from '@/lib/get-tenant';
    import { sql, and, eq, gte } from 'drizzle-orm';
    import type { SQL } from 'drizzle-orm';

    export const runtime = 'nodejs';

    const AGENT_REGEX = /^[a-z0-9_-]{1,32}$/i;

    export async function GET(request: Request) {
      try {
        const { tenantId } = await requireTenantContext();
        const { db: tdb, schema } = getTenantDb(tenantId);

        const url = new URL(request.url);
        const agentParam = url.searchParams.get('agent');
        const agentFilter =
          agentParam && AGENT_REGEX.test(agentParam) ? agentParam.toLowerCase() : null;

        // Optional agent EXISTS filter â€” same pattern as
        // apps/web/app/api/runs/history/route.ts (Plan 16-02 addition).
        const conditions: SQL[] = [];
        if (agentFilter) {
          conditions.push(
            sql`EXISTS (
              SELECT 1 FROM ${schema.events}
              WHERE ${schema.events.runId} = ${schema.runs.id}
                AND lower(${schema.events.agentId}) = ${agentFilter}
            )`,
          );
        }
        const where = conditions.length > 0 ? and(...conditions) : undefined;

        // Start-of-today in the SERVER's local timezone. UTC alignment for the
        // "Completed Today" tile is intentional â€” Phase 16 is a single-region
        // deployment (BeagleHQ Hetzner) and CONTEXT.md doesn't ask for
        // per-tenant timezone awareness.
        const startOfTodayUtc = new Date();
        startOfTodayUtc.setUTCHours(0, 0, 0, 0);

        // 1) totalRuns = count(*) of runs in scope.
        // 2) totalSpendUsd = sum of every event.metadata->>'costUsd' across all
        //    runs in scope. Coalesce to 0.
        // 3) completedRuns = count of runs.status = 'completed' (used to derive
        //    avgCostUsd; not part of the response).
        // 4) completedToday = count of runs.status='completed' AND
        //    runs.updatedAt >= start-of-today.
        //
        // Single SELECT with subqueries is fine: cardinality is small (1 row),
        // tenant runs table is bounded.
        const [row] = await tdb
          .select({
            totalRuns: sql<number>`count(*)::int`,
            totalSpendUsd: sql<number>`coalesce((
              SELECT sum((${schema.events.metadata}->>'costUsd')::numeric)
              FROM ${schema.events}
              WHERE ${schema.events.runId} IN (
                SELECT ${schema.runs.id} FROM ${schema.runs} ${where ? sql`WHERE ${where}` : sql``}
              )
                AND ${schema.events.metadata}->>'costUsd' IS NOT NULL
            ), 0)::float8`,
            completedRuns: sql<number>`sum(case when ${schema.runs.status} = 'completed' then 1 else 0 end)::int`,
            completedToday: sql<number>`sum(case when ${schema.runs.status} = 'completed' and ${schema.runs.updatedAt} >= ${startOfTodayUtc.toISOString()} then 1 else 0 end)::int`,
          })
          .from(schema.runs)
          .where(where);

        const totalRuns = Number(row?.totalRuns ?? 0);
        const totalSpendUsd = Number(row?.totalSpendUsd ?? 0);
        const completedRuns = Number(row?.completedRuns ?? 0);
        const completedToday = Number(row?.completedToday ?? 0);
        const avgCostUsd = completedRuns > 0 ? totalSpendUsd / completedRuns : 0;

        return NextResponse.json(
          {
            totalRuns,
            totalSpendUsd,
            avgCostUsd,
            completedToday,
          },
          { status: 200 },
        );
      } catch (error) {
        console.error('GET /api/runs/history/summary error:', error);
        return NextResponse.json(
          { error: 'Internal server error' },
          { status: 500 },
        );
      }
    }
    ```

    Notes:
    - The "spend" subquery joins back through `runs` so the agent EXISTS filter scopes the cost summation correctly. (When `agent=mo`, only events whose `run_id` belongs to a run that has a Mo event are summed â€” meaning total spend on those runs across ALL agents. Per CONTEXT.md the tile sub-label can read "All agents on Mo's runs" or stay generic; we keep it generic.)
    - `completedToday` uses `runs.updatedAt` as the proxy for completion timestamp because `runs` doesn't have a `completedAt` column today (per Phase 14-02 the hub flips status + updatedAt at completion).
    - Drizzle's `sql` template literal interpolates `${startOfTodayUtc.toISOString()}` as a parameter (NOT string interpolation) â€” Drizzle's tagged-template behavior binds non-fragment values as placeholders.
    - The `${where ? sql\`WHERE ${where}\` : sql\`\`}` pattern conditionally includes the WHERE in the inner subquery to keep the cost computation aligned with the row-level filter.

    B) Create `apps/web/lib/hooks/use-run-history-summary.ts`:
    ```typescript
    'use client';

    import { useQuery } from '@tanstack/react-query';

    export interface RunHistorySummary {
      totalRuns: number;
      totalSpendUsd: number;
      avgCostUsd: number;
      completedToday: number;
    }

    interface SummaryParams {
      agent?: string;
    }

    /**
     * Fetches /api/runs/history/summary. Optionally filtered by agent (same
     * predicate as /api/runs/history accepts).
     */
    export function useRunHistorySummary(params: SummaryParams = {}) {
      return useQuery<RunHistorySummary>({
        queryKey: ['runs', 'history', 'summary', params],
        queryFn: async () => {
          const sp = new URLSearchParams();
          if (params.agent) sp.set('agent', params.agent);
          const qs = sp.toString();
          const url = `/api/runs/history/summary${qs ? `?${qs}` : ''}`;
          const res = await fetch(url);
          if (!res.ok) throw new Error('Failed to fetch run history summary');
          return res.json();
        },
        staleTime: 30_000,
      });
    }
    ```
