    ```bash
    # Replace with operator's actual session cookie.
    COOKIE='Cookie: better-auth.session_token=...'

    # NEW endpoint from 16-01 — operator status check.
    curl -fsS https://console.beaglemind.ai/api/me/operator -H "$COOKIE" | jq .
    # Expected (operator session): { "isOperator": true }
    # Expected (non-operator session): { "isOperator": false }

    # NEW endpoint from 16-03 — Run History summary.
    curl -fsS https://console.beaglemind.ai/api/runs/history/summary -H "$COOKIE" | jq .
    # Expected: { "totalRuns": <int>, "totalSpendUsd": <float>, "avgCostUsd": <float>, "completedToday": <int> }

    # EXTENDED endpoint from 16-02 — agent filter.
    curl -fsS "https://console.beaglemind.ai/api/runs/history?agent=mo&limit=5" -H "$COOKIE" | jq '.runs | length'
    # Expected: 0 or more (a small number because filtered to runs that involved Mo)
    ```
